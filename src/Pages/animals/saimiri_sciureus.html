import altair as alt import pandas as pd from js import myFunction_get, accuracy2 from pyodide.http import open_url node = pd.read_json(open_url("https://raw.githubusercontent.com/diegong21/DashboardTFG/main/animals_con_foto.json")) node = node.drop(node[node['class_id'] != 'Saimiri Sciureus'].index, inplace=False) node = node.drop(node[node['value'] < 3].index, inplace=False) bind = alt.selection_interval(bind='scales') multi = alt.selection_multi(on='mouseover') input_dropdown2 = alt.binding_select(options= [None] + list(node['day'].unique()), labels = ['All'] +list(node['day'].unique()), name='Day: ') selection2 = alt.selection_single(fields=['day'], bind=input_dropdown2) plot = alt.Chart(node).mark_circle( opacity=0.8, stroke='black', strokeWidth=1 ).encode( alt.X('date', title='Hour', scale=alt.Scale(zero=False)), alt.Y('class_id', title='Species'), alt.Size('value', scale=alt.Scale(range=[0, int(myFunction_get('--chart-range'))]), legend=alt.Legend(title='Certainty (%)'), ), color=alt.condition(multi, 'class_id', alt.value('lightgray'), legend=None), tooltip=['image', alt.Tooltip('date', title= 'Date'), alt.Tooltip('hoursminutes(date)', title= 'Hour'), alt.Tooltip('value', title= 'Certainty (%)')] ).add_selection( bind, multi, selection2 ).properties( width=int(myFunction_get('--chart-width')), height=int(myFunction_get('--chart-height')), title=f"Min. Certainty:{accuracy2}" ).transform_filter( selection2 ).configure_axis( labelFontSize=13, titleFontSize=20 ).transform_filter( f"datum.value > {accuracy2}" ) plot
